function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = e.parameter.data || new Date().toLocaleString('pt-BR'); var nome = e.parameter.nome || ''; var whatsapp = e.parameter.whatsapp || ''; var nivel = e.parameter.nivel || ''; var criticos = e.parameter.pontos_criticos || ''; var detalhes = e.parameter.respostas_detalhadas || ''; sheet.appendRow([data, nome, whatsapp, nivel, criticos, detalhes]); return ContentService .createTextOutput(JSON.stringify({ result: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ result: 'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ result: 'ok' })) .setMimeType(ContentService.MimeType.JSON); }
